46double raytracer::Plane::checkRayPlaneIntersection(
47 const Ray &ray, const math::Point3D &position) const
48{
49 const math::Vector3D normal = getNormal(position);
50 const double epsilon = 1e-10;
51 const double rayPlaneAngle = ray.mDirection.dot(normal);
52 const math::Vector3D pointOnPlane = position - ray.mOrigin;
53 const double t = pointOnPlane.dot(normal) / rayPlaneAngle;
54
55 if (abs(rayPlaneAngle) < epsilon) {
56 // Le rayon est parallèle au plan
57 return -1.0;
58 }
59 return t;
60}
