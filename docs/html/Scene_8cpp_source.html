29Color Scene::getLightColor(
30 const IPrimitive &primitive,
31 const math::Point3D &intersectionPoint,
32 const math::Vector3D &normal)
33{
34 Color color = primitive.mBaseColor * _ambient;
35 Color attenuation;
36 double angle = 0.0;
37
38 for (const auto &light : _lights) {
39 angle = normal.dot(math::Vector3D::normalize(
40 light.get()->mBaseOrigin - intersectionPoint));
41
42 if (angle <= 0.0)
43 continue;
44
45 attenuation = light.get()->getColorShadow(intersectionPoint, normal, _primitives);
46
47 color += (attenuation * angle * primitive.mBaseColor * _diffuse);
48 }
49
50 color.clamp();
51 return color;
52}

63Color Scene::getColor(
64 const unsigned int x, const unsigned int y,
65 const unsigned int height, const unsigned int width)
66{
67 Color color(0.0, 0.0, 0.0);
68 math::Point3D intersectionPoint;
69 math::Vector3D normal;
70 const size_t nbPrimitives = _primitives.size();
71 double hitResult = 0.0;
72 double u = 0.0;
73 double v = 0.0;
74
75 // Compute normalized pixel coordinates with jitter for anti-aliasing
76 u = (static_cast<double>(x) + _dis(_gen)) / width;
77 v = (static_cast<double>(y) + _dis(_gen)) / height;
78
79 raytracer::Ray r = _camera.ray(u, v);
80
81 double closestHit = numeric_limits<double>::max();
82 size_t closestPrimitiveIndex = numeric_limits<size_t>::max();
83
84 // Find closest intersected primitive
85 for (size_t j = 0; j < nbPrimitives; j++) {
86 hitResult = _primitives[j].get()->hits(r);
87 if ((hitResult > 0.0) && (hitResult < closestHit)) {
88 closestHit = hitResult;
89 closestPrimitiveIndex = j;
90 }
91 }
92
93 if (closestPrimitiveIndex != numeric_limits<size_t>::max()) {
94 intersectionPoint = r.mOrigin + (r.mDirection * closestHit);
95 normal = math::Vector3D::normalize(
96 _primitives[closestPrimitiveIndex].get()->getNormal(intersectionPoint));
97
98 color += getLightColor(
99 *_primitives[closestPrimitiveIndex],
100 intersectionPoint, normal);
101 }
102
103 color.clamp();
104 return color;
105}

133void Scene::initPlanes(const vector<Plane> &planes,
134 const size_t offset, Factory &factory)
135{
136 const size_t nbPlanes = planes.size();
137 map<string, math::Matrix<double>> matrix;
138 double transformation = 0.0;
139
140 for (size_t i = 0; i < nbPlanes; i++) {
141 transformation = 0.0;
142 _primitives.push_back(factory.createPrimitive(Factory::PrimitiveType::PLANE));
143 _primitives[i + offset].get()->mBaseColor = planes[i].mColor;
144 matrix = planes[i].mAllMatrix;
145 if (planes[i].mAllMatrix.count("translation") > 0)
146 transformation = matrix["translation"](0, 1);
147 _primitives[i + offset].get()->mBasePosition =
148 static_cast<int>(planes[i].mPosition + transformation);
149 _primitives[i + offset].get()->mBaseAxis = planes[i].mAxis;
150 _primitives[i + offset].get()->mBaseAllMatrix = planes[i].mAllMatrix;
151 _primitives[i + offset].get()->mBaseRotation = planes[i].mVectorRotation;

162void Scene::initSpheres(const vector<Sphere> &spheres,

164{
165 const size_t nbSpheres = spheres.size();
166 map<string, math::Matrix<double>> matrix;
167

171 matrix = spheres[i].mAllMatrix;
172 _primitives[i + offset].get()->mBaseCenter = math::Point3D(
173 spheres[i].mX + matrix["translation"](0, 3),
174 spheres[i].mY + matrix["translation"](1, 3),
175 spheres[i].mZ + matrix["translation"](2, 3));
176 } else {
177 _primitives[i + offset].get()->mBaseCenter = math::Point3D(
178 spheres[i].mX, spheres[i].mY, spheres[i].mZ);
179 }
180 _primitives[i + offset].get()->mBaseColor = spheres[i].mColor;
181 _primitives[i + offset].get()->mBaseRadius = spheres[i].mRadius;
182 _primitives[i + offset].get()->mBaseAllMatrix = spheres[i].mAllMatrix;
