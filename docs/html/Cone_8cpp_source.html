114double raytracer::Cone::hits(const Ray &ray)
115{
116 const math::Vector3D rotatedDir = rotateVector(ray.mDirection);
117 const math::Vector3D rotatedCenter = rotateVector(
118 math::Vector3D(ray.mOrigin.mX, ray.mOrigin.mY, ray.mOrigin.mZ));
119 const math::Vector3D oc =
120 math::Point3D(rotatedCenter.mX, rotatedCenter.mY, rotatedCenter.mZ) - mBaseCenter;
121 const double k =
122 static_cast<double>(mBaseRadius) / static_cast<double>(mBaseHeight);
123 const double a = (rotatedDir.dot(rotatedDir))
124 - ((1 + pow(k, 2)) * (rotatedDir.mZ * rotatedDir.mZ));
125 const double b = 2 * ((oc.dot(rotatedDir))
126 - ((1 + pow(k, 2)) * (oc.mZ * rotatedDir.mZ)));
127 const double c = (oc.dot(oc))
128 - ((1 + pow(k, 2)) * (oc.mZ * oc.mZ));
129 const double discriminant = (b * b) - (4 * a * c);
130
131 if (discriminant < 0) {
132 return -1.0;
133 }
134 double t0 = (-b - ::sqrt(discriminant)) / (2.0 * a);
135 double t1 = (-b + ::sqrt(discriminant)) / (2.0 * a);
136 if (t0 > t1) {
137 swap(t0, t1);
138 }
139 const double y0 = rotatedCenter.mZ + (t0 * rotatedDir.mZ);
140 const double y1 = rotatedCenter.mZ + (t1 * rotatedDir.mZ);
141 if (y0 < mBaseCenter.mZ) {
142 if (y1 < mBaseCenter.mZ) {
143 return -1;
144 } else {
145 return t0 + (
146 (t1 - t0) * ((mBaseCenter.mZ - y0) / (y1 - y0))
147 );
148 }
149 } else if (y0 > (mBaseCenter.mZ + mBaseHeight)) {
150 if (y1 > (mBaseCenter.mZ + mBaseHeight)) {
151 return -1;
152 } else {
153 return t0 + (
154 (t1 - t0) * (((mBaseCenter.mZ + mBaseHeight) - y0) / (y1 - y0))
155 );
156 }
157 } else {
158 return t0;
159 }
160}
