139 void Parser::parseRotation(const libconfig::Setting &fileconfigRotation,
140 map<string, math::Matrix<double>> &allMatrix,
141 math::Vector3D &vectorRotation)
142 {
143 double rotationX = 0.0;
144 double rotationY = 0.0;
145 double rotationZ = 0.0;
146 try {
147 const libconfig::Setting &rotation = fileconfigRotation["rotation"];
148 // Récupération de la valeur X
149 if (!rotation.lookupValue("x", rotationX)) {
150 int tmp;
151 if (rotation.lookupValue("x", tmp))
152 rotationX = static_cast<double>(tmp);
153 else exit(84);
154 }
155 // Récupération de la valeur Y
156 if (!rotation.lookupValue("y", rotationY)) {
157 int tmp;
158 if (rotation.lookupValue("y", tmp))
159 rotationY = static_cast<double>(tmp);
160 else exit(84);
161 }
162 // Récupération de la valeur Z
163 if (!rotation.lookupValue("z", rotationZ)) {
164 int tmp;
165 if (rotation.lookupValue("z", tmp))
166 rotationZ = static_cast<double>(tmp);
167 else exit(84);
168 }
169 // Conversion degrés -> radians
170 rotationX = (rotationX * M_PI) / 180.0;
171 rotationY = (rotationY * M_PI) / 180.0;
172 rotationZ = (rotationZ * M_PI) / 180.0;
173
174 // Construction des matrices de rotation
175 allMatrix["rotationX"] = {
176 {1.0, 0.0, 0.0, 0.0},
177 {0.0, std::cos(rotationX), -std::sin(rotationX), 0.0},
178 {0.0, std::sin(rotationX), std::cos(rotationX), 0.0},
179 {0.0, 0.0, 0.0, 1.0}
180 };
181 allMatrix["rotationY"] = {
182 {std::cos(rotationY), 0.0, std::sin(rotationY), 0.0},
183 {0.0, 1.0, 0.0, 0.0},
184 {-std::sin(rotationY), 0.0, std::cos(rotationY), 0.0},
185 {0.0, 0.0, 0.0, 1.0}
186 };
187 allMatrix["rotationZ"] = {
188 {std::cos(rotationZ), -std::sin(rotationZ), 0.0, 0.0},
189 {std::sin(rotationZ), std::cos(rotationZ), 0.0, 0.0},
190 {0.0, 0.0, 1.0, 0.0},
191 {0.0, 0.0, 0.0, 1.0}
192 };
193 }
194 catch (const libconfig::SettingNotFoundException &) {
195 // Si aucune rotation définie, aucune action
196 }
197 vectorRotation = math::Vector3D(rotationX, rotationY, rotationZ);
198 }
