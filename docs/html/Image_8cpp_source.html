70 void Image::displayPPM(const string &filepath)
71 {

73
74 if (!std::filesystem::exists(p.parent_path())) {
75 std::filesystem::create_directories(p.parent_path());

77 ofstream file(filepath, ios::out | ios::trunc);

96 void Image::displaySFML()
97 {
98 sf::Image image;
99 image.create(_width, _height);
100 sf::Texture texture;
101 sf::Sprite sprite;
102 Color color;
103 sf::Uint8 red = 0;
104 sf::Uint8 green = 0;
105 sf::Uint8 blue = 0;
106
107 for (unsigned int y = 0; y < _height; y++) {
108 for (unsigned int x = 0; x < _width; x++) {
109 color = _pixels[make_pair(y, x)];
110 red = static_cast<sf::Uint8>(color.getR() * 255);
111 green = static_cast<sf::Uint8>(color.getG() * 255);
112 blue = static_cast<sf::Uint8>(color.getB() * 255);
113 image.setPixel(x, y, sf::Color(red, green, blue));
114 }
115 texture.loadFromImage(image);
116 sprite.setTexture(texture);
117 _window.get()->clear();
118 _window.get()->draw(sprite);
119 _window.get()->display();
120 }
121 while (_window.get()->isOpen()) {
122 sf::Event event;
123 while (_window.get()->pollEvent(event)) {
124 if (event.type == sf::Event::Closed)
125 _window.get()->close();
126 }
127 }
128 }
