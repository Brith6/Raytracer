27Color PointLight::getColorShadow(
28 const math::Point3D &intersectionPoint,
29 const math::Vector3D &normal,
30 const vector<unique_ptr<IPrimitive>> &primitives)
31{
32 World world;
33 // Direction du rayon allant du point d'intersection vers la source lumineuse
34 const math::Vector3D lightDir = (mBaseOrigin - intersectionPoint);
35 // Création d'un rayon d'ombre légèrement décalé pour éviter les auto-intersections
36 const Ray shadowRay(
37 intersectionPoint + (normal * 0.01),
38 math::Vector3D::normalize(lightDir)
39 );
40 // Recherche de l'intersection la plus proche entre le rayon d'ombre et les primitives
41 const double closestHit = world.hit(shadowRay, primitives);
42
43 // Si une intersection est trouvée avant la lumière, le point est dans l'ombre
44 if (closestHit < lightDir.length()) {
45 return Color(0.0, 0.0, 0.0); // Couleur noire (ombre)
46 }
47 // Sinon, calcul de la couleur de la lumière en fonction de la distance
48 return mBaseColor * world.getDistance(
49 mBaseOrigin, intersectionPoint
50 );
51}
