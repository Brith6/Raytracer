40 std::string axis;

47 if (!setting.lookupValue("axis", axis) ||

54 !setting["color"].lookupValue("g", cg) ||

56 errors.report("Missing or invalid color for plane");

58 }
59 if (axis != "X" && axis != "Y" && axis != "Z") {
60 errors.report("Invalid axis for plane: must be X, Y, or Z");
61 return nullptr;
62 }
63 if (cr < 0 || cr > 255 || cg < 0 || cg > 255 || cb < 0 || cb > 255) {
64 errors.report("Invalid color values for plane: must be between 0 and 255");
65 return nullptr;
66 }
67
68 if (setting.lookupValue("rotationAxis", rotationAxis)) {
69 if (rotationAxis != "X" && rotationAxis != "Y" && rotationAxis != "Z") {
70 errors.report("Invalid rotation axis for plane: must be X, Y, or Z");
71 return nullptr;
72 }
73 if (!setting.lookupValue("rotationAngle", rotationAngle)) {
74 errors.report("Missing rotation angle for plane");
75 return nullptr;
76 }
77 }
78
79 setting.lookupValue("translationX", translationX);
80 setting.lookupValue("translationY", translationY);
81 setting.lookupValue("translationZ", translationZ);
82
83 math::Vector3D rotationVector(0, 0, 0);
84 if (rotationAxis == "X") {
85 rotationVector.x = rotationAngle;
86 } else if (rotationAxis == "Y") {
87 rotationVector.y = rotationAngle;
88 } else if (rotationAxis == "Z") {
89 rotationVector.z = rotationAngle;
90 }
91
92 map<string, math::Matrix<double>> allMatrix;
93 if (!rotationAxis.empty()) {

102 });
103 } else if (rotationAxis == "Y") {
104 double rad = rotationAngle * M_PI / 180.0;
105 rotationMatrix = math::Matrix<double>(4, 4, {
106 cos(rad), 0, sin(rad), 0,
107 0, 1, 0, 0,
108 -sin(rad), 0, cos(rad), 0,
109 0, 0, 0, 1
110 });
111 } else if (rotationAxis == "Z") {
112 double rad = rotationAngle * M_PI / 180.0;
113 rotationMatrix = math::Matrix<double>(4, 4, {
114 cos(rad), -sin(rad), 0, 0,
115 sin(rad), cos(rad), 0, 0,
116 0, 0, 1, 0,
117 0, 0, 0, 1
