58 if (scenef.empty())

65 core::Parser parser(scene._camera);

67 parser.parseFile(scenef);

69 cerr << e.what() << "\n";
70 return 84;

72
73 if (!parser.hasBasicElementInFiles())
74 return 84;

78 static_cast<unsigned int>(scene._camera.height),
79 useSFML
80 );
81

84
85 const std::filesystem::path p(scenef);
